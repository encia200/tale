#include <stdio.h>
#include <set>
#include <utility>
#include <string>
#include <string.h>

using namespace std;

int main()
{
​    int N;
​    char s[51];
​    set<pair<int, string>> a;

​    scanf("%d", &N);

​    while(N--){
​        scanf("%s", s);
​        int len = (int)strlen(s);
​        a.insert({len, string(s)});
​    }

​    for(auto i : a)
​        puts(i.second.data());
}
