# 상자넣기

```c++
#include <stdio.h>
#include <algorithm>

using namespace std;

int N;
int LIS[1001];
int dat[1001];

int main(){
​    scanf("%d", &N);

​    for(int i=1;i<=N;i++)
​        scanf("%d", dat+i);

​    for(int i=1;i<=N;i++){
​        LIS[i] = 1;
​        for(int j=1;j<i;j++){
​            if(dat[i]>dat[j]&&LIS[i]<LIS[j]+1)
​                LIS[i] = LIS[j]+1;
​        }
​    }

​    int res = 1;
​    for(int i=1;i<=N;i++)
​        res = max(res, LIS[i]);
​    printf("%d", res);
​    return 0;
}
```
